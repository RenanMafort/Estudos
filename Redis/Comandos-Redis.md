# comandos-redis

```yaml
#INICIA O REDIS NO LINUX
redis-server --daemonize yes

#PARA O SERVIDOR
redis-cli shutdown

#SERVE PARA ACESSAR TERMINAL DO REDIS DENTRO DO CONTAINER NO DOCKER
docker exec -it <ID_CONTAINER> redis-cli

#SERVE PARA VERIFICAR O TIPO DOS DADOS
TYPE "NOME_CHAVE"

#PARA SETAR(INSERT) VALORES
SET "NOME_CHAVE" VALOR_DA_CHAVE

#PARA PEGAR(SELECT) OS VALORES
GET "NOME_CHAVE"

#PARA ATUALIZAR(UPDATE) OS VALORES
SET "NOME_CHAVE" NOVO_VALOR

#PARA DELETAR(DELET) OS VALORES
DEL "NOME_CHAVE"

#LISTA TODAS AS CHAVES
KEYS *

#LISTA TODAS AS CHAVES QUE INICIA COM "NOTAS"
KEYS NOTAS*

#LISTA TODAS AS CHAVES QUE INICIA COM "NOTAS" E TENHA "MAT" EM SEQUENCIA
KEYS NOTAS*MAT*

#LISTA TODAS AS CHAVES QUE INICIA COM "NOTAS", TENHA "MAT" EM SEQUENCIA E OS "?" INDICA QUANTOS CHAR TEM APÓS O "."
KEYS NOTAS*MAT*NOME.?????

#LISTA TODAS AS CHAVES QUE FOR = O ? INDICA QUANTOS CARACTERES CONTEM DETERMINADA PALAVA APOS O .
KEYS ALGO*QUAL_OUTRA_COISA*NOME.?????

#LISTA TUDO QUE COMEÇA COM "nota: e após : comece com "fi" ou "ma"
keys nota:*:[fima]*

#INSERE VARIAS CHAVES E VALORES
MSET "nota1" 0.2 "nota2" 0.5

######### HASH  ##########
#HSET insere varios valores na mesma chave
HSET "Notas Renan" "mat" 90

#PEGA O VALOR DE DETERMIANDA CHAVE    
HGET "NOME_DO_HASH" "CHAVE"
   

#INSERE VARIOS VALORES NO MESMO HASH
HMSET "Notas Renan" "mat" 90  "fis" 80 ETC...

#LISTA TODAS AS CHAVES E VALORES DO HASH
HGETALL "NOME_DO_HASH"

#EXCLUI A CHAVE E O VALOR
HDELL "NOME_DO_HASH" "CHAVE_DO_HASH"

#SERVE PARA INCREMENTAR O VALOR
HINCRBY "NOME_CHAVE" "VALOR_DO_INCREMENT"
PARA STRINGS É - INCR "NOVA_CHAVE" INCREMENTA DE 1 EM 1, INCRBY "VALOR" INCREMENT DE "VALOR" EM "VALOR"


########LISTAS########
LPUSH "filme" "velozes e furiosos" L = Left push(adicione a esquerda)

RPUSH "filme" "ULTIMO FILME" R = Right push(adicione a direita)

#PEGA O ELEMENTO DE INDICE ..
LINDEX "filme" 0 - INDICE

#LISTA O TAMANHO DA LISTA
LLEN "filme"

#PEGA ELEMENTOS DO 0 A 4
LRANGE 0 4

#PEGA APENAS OS RANGE

############## FILAS ##############
RPUSH "fila:confirma-celular" "91231232131"
ou RPUSH "fila:confirma-celular" "5645645646" "12321321321" "123131312412"

#PEGA O PRIMEIRO DA LISTA, E REMOVE
LPOP "fila:confirma-celular"

#PEGA O ULTIMO DA LISTA, E REMOVE
RPOP "fila:confirma-celular"

#PEGA TODOS DA FILA INDEPENDENTE DO TAMANHO
LRANGE "fila:confirma-celular" 0 -1 #-1 = ultimo, -2 = penultimo

#SERVE PARA PEGAR UM ELEMENTO E REMOVER, CASO A FILA ESTIVER VAZIA, ELE VAI AGUARDAR 20 SEGUNDOS
BLPOP "fila:confirma-celular" 20

#SERVE PARA PEGAR UM ELEMENTO E REMOVER, CASO A FILA ESTIVER VAZIA, COLOCANDO "0" ELE IRÁ AGUARDAR PRA "SEMPRE"
BLPOP "fila:confirma-celular" 0


#SERVE PARA PEGAR UM ELEMENTO E REMOVER, CASO A FILA ESTIVER VAZIA, ELE VAI AGUARDAR 20 SEGUNDOS
BRPOP "fila:confirma-celular" 20

#SERVE PARA PEGAR UM ELEMENTO E REMOVER, CASO A FILA ESTIVER VAZIA, COLOCANDO "0" ELE IRÁ AGUARDAR PRA "SEMPRE"
BRPOP "fila:confirma-celular" 0


#SORTED LIST
ZADD "NOME_CHAVE" "VALOR" "PRA QUEM É O VALOR"

#SERVE PARA LISTAR A QUANTIDADE
ZCARD "NOME_CHAVE"
      
   


```
